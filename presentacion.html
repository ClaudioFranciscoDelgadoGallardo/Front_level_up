Context API - Estado Global

// CarritoContext.js
const CarritoProvider = ({ children }) => {
  const [carrito, setCarrito] = useState([]);

  // Persistencia autom√°tica
  useEffect(() => {
    localStorage.setItem('carrito', JSON.stringify(carrito));
  }, [carrito]);

  const agregarAlCarrito = (producto, qty = 1) => {
    if (producto.stock <= 0) {
      notificar('Sin stock', 'error');
      return;
    }
    
    setCarrito(prev => [...prev, {...producto, qty}]);
  };

  return (
    <CarritoContext.Provider value={{ carrito, agregarAlCarrito }}>
      {children}
    </CarritoContext.Provider>
  );
};

Rutas Protegidas

// ProtectedRoute.js
export default function ProtectedRoute({ children }) {
  const usuario = JSON.parse(
    localStorage.getItem('usuarioActual') || 'null'
  );
  
  const esAdmin = usuario?.rol === 'admin';

  if (!esAdmin) {
    return <Navigate to="/login" replace />;
  }

  return children;
}

Validaciones Reutilizables

// validaciones.js - Funciones puras y testables

export const validarEmailVacio = (email) => {
  return email && email.trim().length > 0;
};

export const validarFormatoEmail = (email) => {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
};

export const validarCredenciales = (credentials) => {
  const usuarios = JSON.parse(
    localStorage.getItem('usuarios') || '[]'
  );
  return usuarios.some(u => 
    u.correo === credentials.email && 
    u.password === credentials.password
  );
};

export const validarStock = (producto, cantidad) => {
  return producto.stock >= cantidad;
};
